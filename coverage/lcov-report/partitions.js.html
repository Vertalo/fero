module.exports = Partitions 
 
function Partitions(peers){
  def(this, 'peers', peers)
 
  peers.cache
    .on('commit', this.change.bind(this))
    .on('change', this.commit.bind(this))
}
 
Partitions.prototype.lookup = function(change){
  return (change.key || change.slice(6, change[5] + 6).toString()).split('.')[0]
}
 
Partitions.prototype.commit = function(change){
  return !this.peers.me && !change.buffer ? this.peers.broadcast(this.peers.serialise(change), this.peers.constants.commands.proxy)
          // on client, create new update = proxy to servers
       :  this.peers.me && !change.buffer ? this.peers.broadcast(this.peers.serialise(this.append(change)), this.peers.constants.commands.commit)
          // on server, create new update => commit & propagate to servers + clients
       :  this.peers.me &&  change.buffer.command == this.peers.constants.commands.proxy  ? this.peers.broadcast(change.buffer, this.peers.constants.commands.commit)
          // on server, committed received update => propagate to peers + clients
       :  this.peers.me &&  change.buffer.command == this.peers.constants.commands.commit ? this.peers.multicast(change.buffer, this.peers.constants.commands.commit)
          // on server, receive commit => propagate to clients
       :  false  
}
 
Partitions.prototype.append = function(change, sender){
  const p = this.lookup(change)
 
  if (!this[p]) this[p] = []
 
  const head = ((last(this[p]).ptime || 0) + 1) % this.peers.constants.partitions.history.max
 
  if (!change.ptime)
    change.ptime = head
  
  if (change.ptime === head) {
    // deb('applying', head, p.grey)
    this[p].push(change)
    return change
  } else if (change.ptime > head)
    sender.nack(p, head)
 
  if (change.ptime < head) {
    deb('reject', head, change.ptime)
  }
}

Partitions.prototype.change = function(buffer, sender){
  const change = buffer.type ? buffer : this.peers.deserialise(buffer) // TODO: Should always be buffer? If not, add test
 
  if (change.type !== 'update' && change.type !== 'remove' && change.type !== 'add') // TODO: bitwise if always buffer
    return deb('non standard change', change, buffer)
 
  set(this.append(change, sender))(this.peers.cache)
  
  return change 
}
